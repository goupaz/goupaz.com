Lesson 30
Author : GOUP
Last Updated : November, 2017

Copyclass Book{
     var $title;
     public $author;

     function __construct($title, $author){
          $this->title = $title;
          $this->author = $author;
     }
};

$book1 = new Book("Harry Potter", "JK Rowling");
